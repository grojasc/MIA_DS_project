caud <- read_csv("caudales estacion manzano.csv")
n<-dim(caud)[1]
caud2 <- caud[15756:n,]
caud3 <-caud2 %>%  mutate(year = year(ymd(dia)), month = month(ymd(dia))) %>% 
  group_by(year, month) %>% summarise(caudal = mean(caudal, na.rm=T))

caud3[is.na(caud3$caudal),] <- c((78.47742+135.36774)/2)
